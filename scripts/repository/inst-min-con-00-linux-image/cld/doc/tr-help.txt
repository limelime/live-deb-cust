# Remove duplicates spaces or double spaces.
	echo "Hello   World" | tr -s ' '

# Remove backspace(BS).
	cat file.txt | tr -d '\b'
	  
# Remove all newlines.
	cat file.txt | tr -d '\n'

# To lowercase.
	echo LOWERCASE | tr '[A-Z]' '[a-z]'
	echo LOWERCASE | tr '[:upper:]' '[:lower:]'
	find . -depth -name '* *' | while IFS= read -r f ; do mv -v "$f" "$(dirname "$f")/$(basename "$f"|tr '[A-Z]' '[a-z]')" ; done
