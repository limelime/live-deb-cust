    [Unit]
    Description=CLD: Systemd Start/Stop service.

    [Service]
    Type=simple
    ExecStart=/usr/local/cld/systemd/cld-systemd-after-boot.sh
    ExecStop=/usr/local/cld/systemd/cld-systemd-before-shutdown-halt-reboot.sh

    [Install]
    WantedBy=multi-user.target

    # Copy CLD.service to /etc/systemd/system/
    # Script called from ExecStart/Stop need to start with #!/bin/bash
    # Enable service: systemctl enable CLD.service
    # Start service: systemctl start CLD.service
    # Check status service: systemctl status CLD.service
